import React from "react";
import Cookies from "js-cookie";
import PropTypes from "prop-types";

/**
 * @typedef {Object} Account
 * @property {number} id - Unique account identifier
 * @property {number} [balance] - Account balance
 * @property {boolean} [closed] - Account status (true = closed)
 * @property {string} [type] - Account type (e.g., Checking, Savings)
 */

/**
 * Modal component for closing a bank account.
 *
 * Displays a confirmation dialog and handles the API request
 * to close the account. Updates local state and triggers a full
 * refresh after successful closure.
 *
 * @component
 * @param {Object} props
 * @param {Account} props.account - The account to be closed
 * @param {function} props.setAccount - Function to update the account state locally
 * @param {function} props.setShowCloseModal - Function to hide/show the modal
 * @param {function} props.refreshAccount - Function to refresh the account from the server
 */
export default function CloseAccountModal({ account, setAccount, setShowCloseModal, refreshAccount }) {

  /**
   * Handles the account closure by calling the API,
   * updating the local state, hiding the modal, and refreshing
   * the account information.
   *
   * @async
   * @function
   */
  const handleCloseAccount = async () => {
    try {
      const token = Cookies.get("access_token");
      const response = await fetch(`http://localhost:8000/accounts/close/${account.id}`, {
        method:"POST",
        headers:{ Authorization:`Bearer ${token}` }
      });

      if (!response.ok) return;

      await response.json();
      // Update local account state
      setAccount({...account, closed:true, balance:0});
      setShowCloseModal(false);
      refreshAccount(); // Full refresh from server
    } catch(e) {
      console.error(e);
    }
  };

  return (
    <div className="modal-overlay">
      <div className="modal-content">
        <h2>Clôturer le compte</h2>
        <p>Le solde sera transféré au compte principal.</p>
        <div className="modal-actions">
          <button className="btn" onClick={() => setShowCloseModal(false)}>Annuler</button>
          <button className="btn danger" onClick={handleCloseAccount}>Confirmer</button>
        </div>
      </div>
    </div>
  );
}

// Prop validation with PropTypes
CloseAccountModal.propTypes = {
  account: PropTypes.shape({
    id: PropTypes.number.isRequired,
    balance: PropTypes.number,
    closed: PropTypes.bool,
    type: PropTypes.string,
  }).isRequired,

  setAccount: PropTypes.func.isRequired,
  setShowCloseModal: PropTypes.func.isRequired,
  refreshAccount: PropTypes.func.isRequired,
};
