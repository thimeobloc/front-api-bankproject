import React, { useState } from "react";
import { useNavigate } from "react-router-dom";
import axios from "axios";
import "../../../src/pages/Beneficiary/Beneficiary.css";
import Cookies from "js-cookie";

/**
 * Page d'inscription d'un nouvel utilisateur.
 *
 * Gère :
 * - La saisie des informations (nom, email, mot de passe, confirmation)
 * - L'appel API pour créer un utilisateur
 * - La gestion du token et des informations utilisateur
 * - La redirection vers la page d'accueil après inscription réussie
 *
 * @component
 */
export default function Signing() {
    const navigate = useNavigate();

    // États locaux pour les champs de saisie
    const [name, setName] = useState("");
    const [email, setEmail] = useState("");
    const [password, setPassword] = useState("");
    const [confirmPassword, setConfirmPassword] = useState("");
    const [error, setError] = useState("");

    /**
     * Gestion de la soumission du formulaire d'inscription
     * @param {Event} e - Événement de soumission
     */
    const handleSubmit = async (e) => {
        e.preventDefault();
        setError("");

        // Vérification que les mots de passe correspondent
        if (password !== confirmPassword) {
            setError("Les mots de passe ne correspondent pas");
            return;
        }

        try {
            // Appel API pour créer un nouvel utilisateur
            const response = await axios.post("http://127.0.0.1:8000/users/", {
                name,
                email,
                password,
            });

            // Stockage du token et de l'ID utilisateur dans les cookies
            Cookies.set("access_token", response.data.access_token, { expires: 1 });
            Cookies.set("user_id", response.data.user.id, { expires: 1 });

            // Stockage des informations utilisateur dans localStorage
            localStorage.setItem(
                "user",
                JSON.stringify({
                    id: response.data.user.id,
                    name: name,
                    email: email
                })
            );

            // Redirection vers la page d'accueil
            navigate("/");
        } catch (err) {
            // Gestion des erreurs détaillées
            if (err.response && err.response.data && err.response.data.detail) {
                const detail = err.response.data.detail;
                if (Array.isArray(detail)) {
                    setError(detail.map(d => d.msg).join(", "));
                } else {
                    setError(detail);
                }
            } else {
                setError("Erreur réseau ou serveur indisponible");
            }
        }
    };

    return (
        <section>
            <div className="center">
                <div className="background">
                    <div className="SignForm">
                        <h1>Inscription</h1>

                        {/* Champs de saisie */}
                        <input
                            className="input_field"
                            type="text"
                            placeholder="Nom"
                            value={name}
                            onChange={(e) => setName(e.target.value)}
                        />
                        <input
                            className="input_field"
                            type="text"
                            placeholder="Adresse Email"
                            value={email}
                            onChange={(e) => setEmail(e.target.value)}
                        />
                        <input
                            className="input_field"
                            type="password"
                            placeholder="Mot de passe"
                            value={password}
                            onChange={(e) => setPassword(e.target.value)}
                        />
                        <input
                            className="input_field"
                            type="password"
                            placeholder="Confirmer le Mot de passe"
                            value={confirmPassword}
                            onChange={(e) => setConfirmPassword(e.target.value)}
                        />

                        {/* Boutons d'action */}
                        <button onClick={handleSubmit}>S'inscrire</button>
                        <button onClick={() => navigate("/Login")}>J'ai déjà un compte</button>

                        {/* Affichage du message d'erreur */}
                        {error && <p style={{ color: "red" }}>{error}</p>}
                    </div>
                </div>
            </div>
        </section>
    );
}
