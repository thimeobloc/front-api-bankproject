import React, { useEffect, useState } from 'react';
import "./Beneficiary.css";
import Cookies from "js-cookie";
import axios from "axios";

import BeneficiaryCard from '../../components/beneficiary/BeneficiaryCard.jsx';
import SearchBar from '../../../src/components/beneficiary/SearchBar';

/**
 * Beneficiary page component.
 *
 * Affiche la liste des comptes de l'utilisateur et ses bénéficiaires enregistrés.
 * Permet de rechercher un bénéficiaire via une barre de recherche et d'afficher
 * dynamiquement les résultats filtrés.
 *
 * @component
 */
export default function Beneficiary() {
  // Liste de tous les bénéficiaires (vide ici, à compléter depuis l'API)
  const beneficiairies = [];

  // État pour la barre de recherche
  const [query, setQuery] = useState("");
  const filteredBeneficiaries = beneficiairies.filter((b) =>
    b.inputName.toLowerCase().includes(query.toLowerCase())
  );

  // État utilisateur et comptes
  const [user, setUser] = useState(null);
  const token = Cookies.get("access_token");
  const [account, setAccounts] = useState([]);

  /**
   * useEffect pour récupérer l'utilisateur et ses comptes depuis l'API.
   * Sauvegarde l'utilisateur depuis localStorage et charge les comptes via axios.
   */
  useEffect(() => {
    const savedUser = localStorage.getItem("user");

    if (savedUser) setUser(JSON.parse(savedUser));

    if (!token) return;

    axios
      .get("http://127.0.0.1:8000/accounts/", {
        headers: {
          Authorization: `Bearer ${token}`,
        },
      })
      .then((res) => {
        setAccounts(res.data);
      })
      .catch((err) => {
        console.error("Erreur de chargement des comptes :", err);
      });
  }, [token]);

  // Si l'utilisateur n'est pas connecté, afficher un message de bienvenue
  if (!user) {
    return (
      <div className="welcome-section">
        <h1>Bienvenue sur Votre Banque</h1>
        <p>Connectez-vous pour accéder à vos comptes ou créez un compte.</p>
      </div>
    );
  }

  return (
    <section>
      <div className="center">
        <h1>Vers quel bénéficiaire ?</h1>

        {/* Barre de recherche */}
        <SearchBar query={query} setQuery={setQuery} />

        <div className="list">
          {filteredBeneficiaries.length === 0 || !query ? (
            <>
              <h2>Vos comptes</h2>
              {account.map((b) => (
                <BeneficiaryCard
                  key={b.id}
                  name={b.inputName}
                  iban={b.iban}
                  imgUrl={b.img}
                />
              ))}

              <h2>Vos Bénéficiaires</h2>
              {beneficiairies.map((b) => (
                <BeneficiaryCard
                  key={b.id}
                  name={b.inputName}
                  iban={b.iban}
                  imgUrl={b.img}
                />
              ))}
            </>
          ) : (
            <>
              <h2>Vous recherchez : "{query}" ...</h2>
              {filteredBeneficiaries.map((b) => (
                <BeneficiaryCard
                  key={b.id}
                  name={b.inputName}
                  iban={b.iban}
                  imgUrl={b.img}
                />
              ))}
            </>
          )}
        </div>
      </div>

      {/* Bouton pour ajouter un nouveau bénéficiaire */}
      <button className="fixed_button">Ajouter un IBAN</button>
    </section>
  );
}
